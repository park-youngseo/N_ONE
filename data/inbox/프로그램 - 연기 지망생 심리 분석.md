defsplit_sentences(text):
"""
한글(text)을문장단위로대략분리합니다.
    - 마침표/물음표/느낌표기준으로단순분리
    - 내용은바꾸지않고, 나누기만합니다.
    """
ifnotisinstance(text, str) ornot text.strip():
return []
# 줄바꿈을공백으로바꾸기
    text = text.replace("\n", " ")
# ., ?, ! 기준으로문장분리 (아주단순한버전)
    sentences = re.split(r"[\.?!]", text)
# 공백제거후빈문자열버리기
return [s.strip() for s in sentences if s.strip()]
