import re
def clean_text(text: str) -> str:
    if not text:
        return ""
    # URL 제거
    text = re.sub(r"http[s]?://\S+", "", text)
    # HTML 태그제거
    text = re.sub(r"<[^>]+>", "", text)
    # 광고/홍보성패턴제거

    text = re.sub(r"(.)\1{2,}", r"\1\1", text)
    # 양쪽공백정리
    text = text.strip()
    return text
